# keepalive

## 环境

```
yum install -y curl gcc openssl-devel libnl3-devel net-snmp-devel
```

## 下载安装

```
wget https://www.keepalived.org/software/keepalived-2.0.20.tar.gz
tar -xf keepalived-2.0.20.tar.gz
cd keepalived-2.0.20
./config --prefix=/usr/local/keepalived/
make && make install 
```

## 配置

```
global_defs {
   router_id node1
}
vrrp_script chk_nginx {
    script "/usr/local/keepalived/check_nginx.sh"
    #15s
    interval 15
    #every check -20
    weight -1
}
vrrp_instance VI_1 {
    state MASTER
    interface em1
    #same in one cluster 
    virtual_router_id 51
    #local-ip
    mcast_src_ip 202.74.2.196
    #master > slave
    priority 100
    #check between master and slave, interval 1s
    advert_int 1
    authentication {
        auth_type PASS
        auth_pass robinregus
    }
    track_script {
        chk_nginx
    }
    virtual_ipaddress {
        202.74.2.200
    }
}
vrrp_instance VI_2 {
    state MASTER
    interface p2p1
    #same in one cluster
    virtual_router_id 52
    #local-ip
    mcast_src_ip 192.168.100.61
    #master > slave
    priority 100
    #check between master and slave, interval 1s
    advert_int 1
    authentication {
        auth_type PASS
        auth_pass robinregus
    }
    track_script {
        chk_nginx
    }
    virtual_ipaddress {
        192.168.100.69
    }
}
```

```
#!/bin/bash -x
#
COUNT1=`ss -anpt | grep nginx | wc -l `
if [ $COUNT1 -eq 0 ] ; then
    killall nginx
    sleep 1
    nginx 
    sleep 2
        COUNT2=`ss -anpt | grep nginx | wc -l`
        if [ $COUNT2 -eq 0 ] ; then
            /usr/bin/kill -15 `cat /var/run/keepalived.pid`
            echo -e "keeplived is stoped"
        else
            exit 0
        fi  
fi
```
